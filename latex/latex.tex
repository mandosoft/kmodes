\begin{algorithm}[H] 
\caption{K Modes Algorithm}

\begin{algorithmic}
\Require{$A_{1} \dots A_{N}$} 
\Ensure{$Clusters$$(C_1, C_2,...C_n)$}
\Statex

\Function{Main}{MSA}
 \State {$C$[$k$] = null}
 \State {Select random $k \gets 1:N$}
\While{$k > 2$} 
 \For{$i \in [0,N)$}
  \State $i \notin k$
  \State distance $G$=0
  \For{$j \in [l,k]$}
  \State $rii\ att[i],att[j]$
  
    \If {$rii > rii \ G$}
     \State $rii\ G = max \ rii$
     \State $assignment[i.l] = j$
     \State $assignment[i.b] = rii \ G$ 
    \EndIf

    \If{$rii\ G > C_n[b]$}
     \State set $ C_n$ index and value 
    \EndIf
\EndFor
\EndFor
\EndWhile
\State \Return {$Clusters$}
\EndFunction
\end{algorithmic}
\end{algorithm}
